class MightyStrike : Skill, IUseSkill
{

    public MightyStrike() : base("Потужний удар", 25)
    {
        skillNumber = 100;
    }

    public bool use(Player player)
    {
        if (!decreaseEp(player))
        {
            return false;
        }

        player.BonusDamage = 30;

        return true;
    }
}
