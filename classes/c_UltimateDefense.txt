class UltimateDefence : Skill, IUseSkill
{
    public UltimateDefence() : base("Ультимативний захист", 50)
    {
        skillNumber = 200;
    }

    public bool use(Player player)
    {
        if (!decreaseEp(player))
        {
            return false;
        }

        player.BonusProtect = 80;

        return true;
    }
}
