class RegenerateHp : Skill, IUseSkill
{
    public RegenerateHp() : base("Регенерація", 80)
    {
        skillNumber = 300;
    }

    public bool use(Player player)
    {
        if (!decreaseEp(player))
        {
            return false;
        }

        player.addedHealth(250);

        return true;
    }
}
