class Weapon : NameTrait
{
    private int v;

    public int Damage { get; set; }
    public int Price { get; set; }

    public Weapon(string name, int damage, int price) : base(name) //,int price
    {
        Damage = damage;
        Price = price;
    }

    public Weapon(string name, int v) : base(name)
    {
        this.v = v;
    }
}
